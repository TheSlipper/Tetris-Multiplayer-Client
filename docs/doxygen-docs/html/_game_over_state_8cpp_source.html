116  void GameOverState::parseResponseData(const std::string &data)
117  {
118  ArktisEngine::StringTokenizer tokenizer{data};
119  int iterator = 0;
120  bool p1Data = true;
121  for (const auto &t : tokenizer)
122  {
123  if (iterator < 3)
124  iterator++;
125  else if (p1Data)
126  {
127  if (iterator < 10)
128  this->assignDataByIterator(iterator, t, this->_data->userData);
129  else if (iterator == 10)
130  this->p1Pts = std::stof(t);
131  else if (iterator == 11)
132  this->p1Time = std::stol(t);
133  p1Data = false;
134  }
135  else
136  {
137  if (iterator < 10)
138  this->assignDataByIterator(iterator, t, this->opponentData);
139  else if (iterator == 10)
140  this->p2Pts = std::stof(t);
141  else if (iterator == 11)
142  this->p2Time = std::stol(t);
143  iterator++;
144  p1Data = true;
145  }
146  }
