205  void HomeScreenState::loadArticles()
206  {
207  this->_data->messaging.SendStringData("GET_UPDATE_LOGS");
208  std::string response = this->_data->messaging.GetStringResponse();
209  boost::char_separator<char> sep{"\r\n"};
210  ArktisEngine::StringTokenizer tokenizer{response, sep};
211  int i = -1;
212  for (const std::string &token : tokenizer)
213  {
214  switch (i)
215  {
216  case 0:
217  this->addArticleHeader(token);
218  break;
219  case 1:
220  this->addArticleContent(token);
221  break;
222  case 2:
223  this->addArticleAuthor(token);
224  break;
225  case 3:
226  this->addArticleDate(token);
227  break;
228  case -1:
229  const std::string codename = "SEND_UPDATE_LOGS";
230  if (codename.compare(token) != 0)
231  return;
232  break;
233  }
234  i = (i == 3) ? 0 : i + 1;
235  }
236  }
