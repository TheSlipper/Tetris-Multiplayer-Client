52  void LoginState::HandleInput()
53  {
54  sf::Event event;
55 
56  while(this->_data->window.pollEvent(event))
57  {
58  if (sf::Event::Closed == event.type)
59  {
60  this->_data->messaging.DisconnectSocket();
61  this->_data->window.close();
62  }
63  else if (sf::Event::MouseButtonPressed == event.type)
64  {
65  this->_passwordInput->SetFocus(false);
66  this->_loginInput->SetFocus(false);
67 
68  if (this->_data->input.IsInputFieldClicked(*this->_loginInput, sf::Mouse::Left, this->_data->window))
69  this->_loginInput->SwitchFocusState();
70  else if (this->_data->input.IsInputFieldClicked(*this->_passwordInput, sf::Mouse::Left, this->_data->window))
71  this->_passwordInput->SwitchFocusState();
72  else if (this->_data->input.IsButtonClicked(*this->_loginButton, sf::Mouse::Left, this->_data->window))
73  this->sendFormData();
74  }
75  else if (sf::Event::TextEntered == event.type)
76  {
77  char inputCharacter = static_cast<char>(event.text.unicode);
78  if (inputCharacter == (char)8)
79  {
80  // Backspace
81  if (this->_loginInput->HasFocus())
82  this->_loginInput->RemoveLastContentChar();
83  else if (this->_passwordInput->HasFocus() && this->passwd.length() > 0)
84  {
85  this->passwd.pop_back();
86  this->_passwordInput->RemoveLastContentChar();
87  }
88  }
89  else if (inputCharacter == (char)9)
90  {
91  // Tab
92  if (this->_loginInput->HasFocus())
93  {
94  this->_loginInput->SwitchFocusState();
95  this->_passwordInput->SetFocus(true);
96  }
97  else
98  {
99  this->_loginInput->SetFocus(true);
100  this->_passwordInput->SwitchFocusState();
101  }
102  }
103  else if (inputCharacter == (char)10 || inputCharacter == (char)13) // Enter
104  this->sendFormData();
105  else
106  {
107  if (this->_loginInput->HasFocus())
108  this->_loginInput->AppendCharacter(inputCharacter);
109  else if (this->_passwordInput->HasFocus())
110  {
111  this->passwd += inputCharacter;
112  this->_passwordInput->AppendCharacter('*');
113  }
114  }
115  }
116  }
117  }
